function drawCircle(center, radius, outlinecolor, mynation) {
	if(oldPoly) {
		map.removeOverlay(oldPoly);
		oldPoly=null;
	}
	
	var d2r = Math.PI/180; 
      var r2d = 180/Math.PI; 
      var Clat = (radius/3963)*r2d; 
      var Clng = Clat/Math.cos(center.lat()*d2r); 
      var points = []; 
      for (var i=0; i < 33; i++) { 
        var theta = Math.PI * (i/16); 
        Cy = center.lat() + (Clat * Math.sin(theta)); 
        Cx = center.lng() + (Clng * Math.cos(theta)); 
        var P = new GPoint(Cx,Cy); 
        points.push(P); 
      }


	//var outlinecolor 
	//outlinecolor="#000000"
	
	var Cwidth = 4;
	points.push(points[0]);	// close the circle
	if (mynation == 1){
	oldPoly = new GPolygon(points,outlinecolor,Cwidth,1,outlinecolor,.1)
	map.addOverlay(oldPoly);
	}
	else {
	theirPoly = new GPolygon(points,outlinecolor,Cwidth,1,outlinecolor,.1)
	map.addOverlay(theirPoly);
	}	
}
